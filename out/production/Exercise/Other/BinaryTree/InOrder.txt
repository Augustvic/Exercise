public class InOrder {
    public void inOrder(BinaryTreeNode root) {
        if (root == null)
            return;
        Stack<BinaryTreeNode> s = new Stack<>();
        while (root != null) {
            while (root != null) {
                s.push(root);
                root = root.left;
            }
            BinaryTreeNode root = s.pop();
            System.out.println(root.val);
            if (root.right != null) {
                s.push(root.right);
                root = root.right;
            }
        }
    }
}
