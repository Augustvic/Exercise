public class P566 {
    public int[][] matrixReshape(int[][] nums, int r, int c) {
        int m = nums.length;
        int n = nums[0].length;
        if (m * n != r * c)
            return nums;
        int[][] reshapeMatrix = new int[r][c];
        int index = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                reshapeMatrix[index / c][index % c] = nums[i][j];
                index++;
            }
        }
        return reshapeMatrix;
    }
}
